PREFIX ns: <http://www.owl-ontologies.com/stations-velos.owl#>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>

SELECT ?station ?n ?v
WHERE {
	 ?station rdf:type ns:station.
	 ?station ns:name ?n.
	 ?station ns:ville ?v.
	 FILTER regex(?v, "%PARAMETER%").
}
